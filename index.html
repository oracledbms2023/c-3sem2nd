            public:

                void sub();
                void mult();
                void display();
            };
            
            void result::read() {
                cout  "\nEnter the order of matrix A: ";

                for (i = 0; i less m1; i++) {
                    for (j = 0; j less n1; j++) {

                    cout  "\n";
                }
            }
            
            void result::sum() {
                if ((m1 != m2) || (n1 != n2)) {
                    cout  "The order should be the same for addition";
                } else {
                    for (i = 0; i less m1; i++) {
                        for (j = 0; j less n1; j++) {
                            c[i][j] = a[i][j] + b[i][j];
                        }
                    }
                }
            }

                if ((m1 != m2) || (n1 != n2)) {
                    cout  "The order should be the same for subtraction ";
                } else {
                    for (i = 0; i less m1; i++) {
                        for (j = 0; j less n1; j++) {
                            c[i][j] = a[i][j] - b[i][j];
                        }
                    }
                }
            }

                if (n1 != m2) {
                    cout  "Invalid order limit ";
                } else {
                    for (i = 0; i less m1; i++) {
                        for (j = 0; j less n2; j++) {
                            c[i][j] = 0;
                            for (int k = 0; k less n1; k++) {
                                c[i][j] += a[i][k] * b[k][j];

                        }
                    }
                }
            }

            #include iostream>
            using namespace std;
            
            template class T>
            void bubble(T a[], int n) {
                int i, j;
                for(i = 0; i less n - 1; i++) {
                    for(j = 0; j less n - 1; j++) {
                        if(a[j] > a[j + 1]) {
                            T temp;
                            temp = a[j];
                            a[j] = a[j + 1];
                            a[j + 1] = temp;
                        }
                    }
                }
            }
            
            int main() {
                int a[6] = {99, 58, 75, 33, 29, 11};
                char b[4] = {'z', 'f', 'x', 'a'};
            
                bubble(a, 6);
                cout  "\nSorted Order Integers: ";
                for(int i = 0; i less 6; i++)
                    cout  a[i]  " ";
            
                bubble(b, 4);
                cout  "\nSorted Order Characters: ";
                for(int j = 0; j less 4; j++)
                    cout  b[j]  " ";
            
                return 0;
            }

            }
        }
